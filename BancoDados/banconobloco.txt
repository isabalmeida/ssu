CREATE TABLE Pessoa (
    id_pessoa int PRIMARY KEY,
    nome varchar(45),
    cpf varchar(16),
    telefone varchar(21),
    data_nascimento date,
    genero char,
    email varchar(45),
    senha varchar(12),
    endereco varchar(45),
    bairro varchar(25),
    numero varchar(6)
);

CREATE TABLE MEDICO (
    id_medico int,
    crm varchar(15),
    FK_Pessoa_id_pessoa int,
    FK_ESPECIALIDADE_id_especialidade int,
    PRIMARY KEY (id_medico, FK_Pessoa_id_pessoa)
);

CREATE TABLE PACIENTE (

    FK_Pessoa_id_pessoa int,

);

CREATE TABLE SECRETARIA (
